unset multiplot

set term wxt 1

set multiplot layout 2,1 

set title "Prices"
set xdata time
set timefmt "%H:%M:%S"
set format x "%H:%M:%.3S"
set xtics out rotate by 0
set xlabel "Time"
set ylabel "Price"
set y2label "Alpha"
set autoscale y
set autoscale y2
set x2zeroaxis
set grid
set xrange ["14:00:00" : "18:00:00"]
set y2range [-2 : 2]
plot \
    "si.txt" using 1:2 title 'Spot' with lines axes x1y1, \
    "si.txt" using 1:3 title 'Price' with lines axes x1y1, \
    "si.txt" using 1:4 title 'Price-Open' with lines axes x1y1, \
    "si.txt" using 1:9 title 'Alpha-Open' with lines axes x1y2

set title "Moving average trend"
set xdata time
set timefmt "%H:%M:%S"
set format x "%H:%M:%.3S"
set xtics out rotate by 0
set xlabel "Time"
set ylabel "Spot"
set y2label "Trend"
set autoscale y
set autoscale y2
set y2tics
set xzeroaxis
set x2zeroaxis
set grid
set xrange ["14:00:00" : "18:00:00"]
plot \
    "si.txt" using 1:2 title 'Spot' with lines axes x1y1, \
    "si.txt" using 1:5 title 'Trend' with lines axes x1y2
