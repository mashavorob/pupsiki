unset multiplot

fname = "si-10.txt"
from = "11:00:00"
to   = "14:00:00"

set term wxt 0

set multiplot layout 2,1 

set title "Prices"
set xdata time
set timefmt "%H:%M:%S"
set format x "%H:%M:%.3S"
set xtics out rotate by 0
set xlabel "Time"
set ylabel "Price"
set y2label "Alpha"
set autoscale y
set autoscale y2
set x2zeroaxis
set grid
set xrange [from : to]
set y2range [-2 : 2]
plot \
    fname using 1:2 title 'Spot' with lines axes x1y1, \
    fname using 1:3 title 'Price' with lines axes x1y1, \
    fname using 1:4 title 'Price-Open' with lines axes x1y1, \
    fname using 1:9 title 'Alpha-Open' with lines axes x1y2

set title "Traded volume"
set xdata time
set timefmt "%H:%M:%S"
set format x "%H:%M:%.3S"
set xtics out rotate by 0
set xlabel "Time"
set ylabel "Spot"
set y2label "Volume"
set autoscale y
set autoscale y2
set y2tics
set xzeroaxis
set x2zeroaxis
set grid
set xrange [from : to]
