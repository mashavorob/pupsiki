#set terminal wxt
set multiplot layout 2,1 

set title "Prices"
set xdata time
set timefmt "%H:%M:%S"
set format x "%H:%M:%.3S"
set xtics out rotate by 0
set xlabel "Time"
set ylabel "Price"
set y2label "Alpha"
set autoscale y
set autoscale y2
set x2zeroaxis
set grid
set xrange ["14:00:00" : "18:00:00"]
set y2range [-2 : 2]
plot \
    "si.txt" using 1:2 title 'Spot' with lines axes x1y1, \
    "si.txt" using 1:3 title 'Price1' with lines axes x1y1, \
    "si.txt" using 1:4 title 'Price2' with lines axes x1y1, \
    "si.txt" using 1:7 title 'Alpha' with lines axes x1y2
